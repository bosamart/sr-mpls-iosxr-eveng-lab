! ====================================================================
! CE3 - Customer Edge  (Cisco CSR 1000v / IOS-XE)  -- AS 65001
! LIGHT drop-in replacement for CE1 (XRv9000) to save lab RAM (~4 GB vs 8 GB).
! Same addressing as CE1 — the PEs are UNCHANGED and never know the difference.
! >>> Run CE3/CE4 OR CE1/CE2, never both at once (shared PE ports + RAM). <<<
! L3VPN : eBGP to R1 (192.168.11.1) on Gi1 ; Loopback 11.11.11.11/32
! L2VPN : Gi2 in shared subnet 172.16.0.0/24 (EVPN-VPWS to CE4)
! Cabling: CSR Gi1 -> R1 Gi0/0/0/0   |   CSR Gi2 -> R1 Gi0/0/0/2
! Notes  : IOS-XE has NO 'route-policy PASS' requirement (that's XR-only);
!          'network' statements advertise to eBGP directly.
! ====================================================================
hostname CE3
!
interface Loopback0
 ip address 11.11.11.11 255.255.255.255
!
interface GigabitEthernet1
 description L3VPN link to R1 (R1 Gi0/0/0/0)
 ip address 192.168.11.2 255.255.255.252
 no shutdown
!
interface GigabitEthernet2
 description L2VPN (EVPN-VPWS) segment to CE4
 ip address 172.16.0.1 255.255.255.0
 no shutdown
!
router bgp 65001
 bgp router-id 11.11.11.11
 bgp log-neighbor-changes
 network 11.11.11.11 mask 255.255.255.255
 neighbor 192.168.11.1 remote-as 100
!
end
