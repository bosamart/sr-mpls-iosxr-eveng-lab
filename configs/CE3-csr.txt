! Light (~4 GB) and always-available — does NOT touch CE1/CE2 or the
! existing topology. Power on the CEs each test needs; RAM = powered nodes only.
! L3VPN : eBGP to R1 (192.168.30.1) on Gi1 ; Loopback 33.33.33.33/32
! Reaches CE4 (44.44.44.44) and CE1/CE2 across the SR core (same VRF).
! Cabling: CSR Gi1 -> R1 Gi0/0/0/4   (new port — set R1 to 5 interfaces in EVE-NG)

! Notes : IOS-XE advertises 'network' to eBGP directly (no XR route-policy needed).

hostname CE3
!
interface Loopback0
 ip address 33.33.33.33 255.255.255.255
!
interface GigabitEthernet1
 description L3VPN link to R1 (R1 Gi0/0/0/4)
 ip address 192.168.30.2 255.255.255.252
 no shutdown
!
router bgp 65003
 bgp router-id 33.33.33.33
 bgp log-neighbor-changes
 network 33.33.33.33 mask 255.255.255.255
 neighbor 192.168.30.1 remote-as 100
!
end
