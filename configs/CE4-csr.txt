! ====================================================================
! CE4 - Customer Edge  (Cisco CSR 1000v / IOS-XE)  -- AS 65002
! LIGHT drop-in replacement for CE2 (XRv9000) to save lab RAM (~4 GB vs 8 GB).
! Same addressing as CE2 — the PEs are UNCHANGED and never know the difference.
! >>> Run CE3/CE4 OR CE1/CE2, never both at once (shared PE ports + RAM). <<<
! L3VPN : eBGP to R4 (192.168.44.1) on Gi1 ; Loopback 22.22.22.22/32
! L2VPN : Gi2 in shared subnet 172.16.0.0/24 (EVPN-VPWS to CE3)
! Cabling: CSR Gi1 -> R4 Gi0/0/0/1   |   CSR Gi2 -> R4 Gi0/0/0/0
! Notes  : IOS-XE has NO 'route-policy PASS' requirement (that's XR-only).
! ====================================================================
hostname CE4
!
interface Loopback0
 ip address 22.22.22.22 255.255.255.255
!
interface GigabitEthernet1
 description L3VPN link to R4 (R4 Gi0/0/0/1)
 ip address 192.168.44.2 255.255.255.252
 no shutdown
!
interface GigabitEthernet2
 description L2VPN (EVPN-VPWS) segment to CE3
 ip address 172.16.0.2 255.255.255.0
 no shutdown
!
router bgp 65002
 bgp router-id 22.22.22.22
 bgp log-neighbor-changes
 network 22.22.22.22 mask 255.255.255.255
 neighbor 192.168.44.1 remote-as 100
!
end
