! ADDITIONAL L3VPN site in VRF CUST-A, on its OWN PE port + IPs.
! Light (~4 GB) and always-available — does NOT touch CE1/CE2 or the
! existing topology. Ideal Phase 8 trigger: it always advertises 44.44.44.44,
! R4 colors it, and that drives ODN on R1.
! L3VPN : eBGP to R4 (192.168.40.1) on Gi1 ; Loopback 44.44.44.44/32
! Reaches CE3 (33.33.33.33) and CE1/CE2 across the SR core (same VRF).
! Cabling: CSR Gi1 -> R4 Gi0/0/0/4   (new port — set R4 to 5 interfaces in EVE-NG)
! L2VPN : none — EVPN-VPWS stays CE1<->CE2 (CSR CEs are L3-only here).

hostname CE4
!
interface Loopback0
 ip address 44.44.44.44 255.255.255.255
!
interface GigabitEthernet1
 description L3VPN link to R4 (R4 Gi0/0/0/4)
 ip address 192.168.40.2 255.255.255.252
 no shutdown
!
router bgp 65004
 bgp router-id 44.44.44.44
 bgp log-neighbor-changes
 network 44.44.44.44 mask 255.255.255.255
 neighbor 192.168.40.1 remote-as 100
!
end
